Q1:Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype

        // Parent object
function Parent(name) {
  this.name = name;
}

Parent.prototype.greet = function() {
  console.log(`Hello, my name is ${this.name}.`);
};

// Child object
function Child(name, age) {
  Parent.call(this, name);
  this.age = age;
}

Child.prototype = Object.create(Parent.prototype);
Child.prototype.constructor = Child;

// Creating instances of parent and child objects
const parent = new Parent("John");
const child = new Child("Jane", 10);

// Using the methods from parent's prototype in child object
parent.greet(); 
child.greet(); 

output:
Hello, my name is John.
Hello, my name is Jane.

Q3:Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays

    Array.prototype.calculateSum = function() {
        let sum = 0;
        for (let i = 0; i < this.length; i++) {
            sum += this[i];
          }
          return sum;
        } 
        const array1 = [11, 22, 33, 44, 55];
        const array2 = [12, 23, 32, 41, 54]; 
    const sum1 = array1.calculateSum();
    const sum2 = array2.calculateSum();
    console.log(sum1);
    console.log(sum2);
    
    output:
    165
    162
